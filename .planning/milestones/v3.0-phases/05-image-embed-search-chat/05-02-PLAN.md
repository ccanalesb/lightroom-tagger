```python
from __future__ import annotations

import numpy as np
import sqlite_vec
from PIL import Image
from sentence_transformers import SentenceTransformer

CLIP_EMBED_MODEL_ID = "clip-ViT-B-32"
CLIP_EMBED_DIM = 512

_clip_model: SentenceTransformer | None = None


def _get_clip_model() -> SentenceTransformer:
    global _clip_model
    if _clip_model is None:
        _clip_model = SentenceTransformer(CLIP_EMBED_MODEL_ID)
    return _clip_model


def encode_images(paths: list[str], *, batch_size: int = 8) -> np.ndarray:
    """Load images from paths, return float32 array of shape (N, CLIP_EMBED_DIM)."""
    ...


def encode_text_for_clip(texts: list[str], *, batch_size: int = 24) -> np.ndarray:
    """Encode text strings in CLIP space; shape (N, CLIP_EMBED_DIM)."""
    ...
```

Implementation requirements:

- `encode_images`: for each path in batches of `batch_size`, open with `Image.open(path).convert("RGB")` (caller/handler may catch failures; this function may let `OSError` propagate or document that paths must exist — prefer **no** broad try/except here). Collect PIL images in a list, then call `_get_clip_model().encode(images, batch_size=batch_size, normalize_embeddings=True, show_progress_bar=False, convert_to_numpy=True)`. Return `np.asarray(raw, dtype=np.float32)` with shape `(N, CLIP_EMBED_DIM)`.
- `encode_text_for_clip`: call `_get_clip_model().encode(texts, batch_size=batch_size, normalize_embeddings=True, show_progress_bar=False, convert_to_numpy=True)`, return `np.asarray(..., dtype=np.float32)`.
- `numpy_to_clip_vec_blob(arr: np.ndarray) -> bytes`: assert `arr.shape[-1] == CLIP_EMBED_DIM` (handle `(512,)` and `(1, 512)` by squeezing if needed, same pattern as `numpy_to_vec_blob` in `embedding_service.py`). Return `sqlite_vec.serialize_float32(arr.astype(np.float32).flatten().tolist())`.
</action>
