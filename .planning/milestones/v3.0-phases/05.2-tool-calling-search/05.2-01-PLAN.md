# Plan 05.2-01 — Backend tool infrastructure

## Objective
Create the tool-calling backbone: typed tool schemas (JSON Schema), a synchronous tool executor, and an LLM conversation loop that handles tool_calls responses and feeds results back before generating the final assistant message.

## Context
Currently `run_nl_catalog_filter_llm_multi_turn` does a single-shot LLM call and returns raw JSON. We need to replace this with a multi-turn loop: send tools → receive tool_calls → execute tools → send tool results → receive final response.

The existing `query_catalog_images` and related DB helpers in `database.py` remain unchanged — the tool executor delegates to them.

## Tasks

```python
"""Tool schemas and executor for LLM function-calling search."""
from __future__ import annotations
import sqlite3
from typing import Any

# --- Tool schemas (OpenAI function-calling format) ---

SEARCH_CATALOG_TOOL = {
    "type": "function",
    "function": {
        "name": "search_catalog",
        "description": (
            "Search the photo catalog with optional filters. "

            "type": "object",

                    "type": "string",

                    "type": "string",

                },

                    "type": "integer",

                    "maximum": 10,
                    "description": "Minimum score (1–10). Requires score_perspective.",
                },
                "min_rating": {
                    "type": "integer",
                    "minimum": 1,
                    "maximum": 5,
                    "description": "Minimum Lightroom star rating.",
                },
                "limit": {
                    "type": "integer",
                    "minimum": 1,
                    "maximum": 100,
                    "description": "Number of results to return. Default 10. Use 1 for 'the best'.",
                },
            },
            "additionalProperties": False,
        },
    },
}

GET_SCORING_PERSPECTIVES_TOOL = {
    "type": "function",
    "function": {
        "name": "get_scoring_perspectives",
        "description": "Returns available scoring perspective slugs and descriptions. Call this before using score_perspective in search_catalog.",
        "parameters": {"type": "object", "properties": {}, "additionalProperties": False},
    },
}

FILTER_BY_DATE_TOOL = {
    "type": "function",
    "function": {
        "name": "filter_by_date",
        "description": "Filter images by date range.",
        "parameters": {
            "type": "object",
            "properties": {
                "date_from": {"type": "string", "description": "Start date YYYY-MM-DD"},
                "date_to": {"type": "string", "description": "End date YYYY-MM-DD"},
                "sort_direction": {"type": "string", "enum": ["newest", "oldest"]},
                "limit": {"type": "integer", "minimum": 1, "maximum": 100},
            },
            "additionalProperties": False,
        },
    },
}

ALL_TOOLS = [SEARCH_CATALOG_TOOL, GET_SCORING_PERSPECTIVES_TOOL, FILTER_BY_DATE_TOOL]


def execute_tool(name: str, args: dict[str, Any], db: sqlite3.Connection) -> dict[str, Any]:
    """Execute a tool call and return a result dict suitable for a tool message."""
    if name == "get_scoring_perspectives":
        return _exec_get_scoring_perspectives(db)
    if name == "search_catalog":
        return _exec_search_catalog(args, db)
    if name == "filter_by_date":
        return _exec_filter_by_date(args, db)
    return {"error": f"Unknown tool: {name}"}


def _exec_get_scoring_perspectives(db: sqlite3.Connection) -> dict[str, Any]:
    rows = db.execute(
        "SELECT DISTINCT perspective_slug FROM image_scores WHERE is_current = 1 ORDER BY perspective_slug"
    ).fetchall()
    slugs = [r["perspective_slug"] for r in rows]
    return {"perspectives": slugs}


def _exec_search_catalog(args: dict[str, Any], db: sqlite3.Connection) -> dict[str, Any]:
    from lightroom_tagger.core.database import query_catalog_images

    limit = min(int(args.get("limit") or 10), 100)

    kwargs: dict[str, Any] = {}
    if args.get("description_search"):
        kwargs["description_search"] = args["description_search"]
    if args.get("score_perspective"):
        kwargs["score_perspective"] = args["score_perspective"]
    if args.get("sort_by_score"):
        kwargs["sort_by_score"] = args["sort_by_score"]
    if args.get("sort_by_date"):
        kwargs["sort_by_date"] = args["sort_by_date"]
    if args.get("min_score") is not None:
        kwargs["min_score"] = int(args["min_score"])
    if args.get("min_rating") is not None:
        kwargs["min_rating"] = int(args["min_rating"])

    rows, total = query_catalog_images(db, limit=limit, offset=0, **kwargs)
    images = _rows_to_tool_result(rows)
    return {"total_matched": total, "returned": len(images), "images": images}


def _exec_filter_by_date(args: dict[str, Any], db: sqlite3.Connection) -> dict[str, Any]:
    from lightroom_tagger.core.database import query_catalog_images

    limit = min(int(args.get("limit") or 10), 100)
    kwargs: dict[str, Any] = {}
    if args.get("date_from"):
        kwargs["date_from"] = args["date_from"]
    if args.get("date_to"):
        kwargs["date_to"] = args["date_to"]
    if args.get("sort_direction"):
        kwargs["sort_by_date"] = args["sort_direction"]

    rows, total = query_catalog_images(db, limit=limit, offset=0, **kwargs)
    images = _rows_to_tool_result(rows)
    return {"total_matched": total, "returned": len(images), "images": images}


def _rows_to_tool_result(rows: list) -> list[dict[str, Any]]:
    """Convert DB rows to rich tool result dicts."""
    result = []
    for row in rows:
        item: dict[str, Any] = {
            "key": row.get("key") or row.get("id"),
            "filename": row.get("filename"),
            "date_taken": row.get("date_taken"),
        }
        if row.get("score") is not None:
            item["score"] = row["score"]
        if row.get("score_perspective"):
            item["score_perspective"] = row["score_perspective"]
        if row.get("score_rationale"):
            item["score_rationale"] = row["score_rationale"]
        if row.get("description") or row.get("summary"):
            item["description"] = row.get("description") or row.get("summary")
        if row.get("mood_tags"):
            item["mood_tags"] = row["mood_tags"]
        if row.get("dominant_colors"):
            item["dominant_colors"] = row["dominant_colors"]
        result.append(item)
    return result
```

### Task 2 — Add `run_tool_calling_search` to `nl_catalog_search.py`

Add a new function that runs the full tool-calling loop. Add after the existing `run_nl_catalog_filter_llm_multi_turn` function:

```python
def run_tool_calling_search(
    messages: list[dict],
    *,
    provider_id: str | None = None,
    model: str | None = None,
    db: "sqlite3.Connection",
    log_callback: LogCallback = None,
    max_tool_rounds: int = 5,
) -> tuple[str, list[dict]]:
    """Run multi-turn tool-calling search loop.

    Returns (assistant_text, updated_messages_with_tool_turns).
    The returned messages list includes tool_call and tool_result turns
    so the frontend can persist them for multi-turn continuity.
    """
    import json
    from lightroom_tagger.core.search_tools import ALL_TOOLS, execute_tool

    registry = ProviderRegistry()
    desc_defaults = registry.defaults.get("description", {}) or {}
    resolved_provider = provider_id or desc_defaults.get("provider")
    if not resolved_provider:
        raise ModelUnavailableError(
            "No provider configured for tool-calling search",
            provider=None,
            model=None,
        )
    resolved_model = model or desc_defaults.get("model")
    if not resolved_model:
        models = registry.list_models(resolved_provider)
        if not models:
            raise ModelUnavailableError(
                f"No models for provider {resolved_provider!r}",
                provider=resolved_provider,
                model=None,
            )
        resolved_model = models[0]["id"]

    client = registry.get_client(resolved_provider)
    conv = list(_normalize_nl_messages(messages))
    updated_messages: list[dict] = list(messages)

    for _round in range(max_tool_rounds):
        response = client.chat.completions.create(
            model=resolved_model,
            messages=conv,
            tools=ALL_TOOLS,
            tool_choice="auto",
        )
        msg = response.choices[0].message

        # No tool calls → final text response
        if not msg.tool_calls:
            text = msg.content or ""
            updated_messages.append({"role": "assistant", "content": text})
            return text, updated_messages

        # Execute tool calls
        tool_calls_payload = [
            {
                "id": tc.id,
                "type": "function",
                "function": {"name": tc.function.name, "arguments": tc.function.arguments},
            }
            for tc in msg.tool_calls
        ]
        conv.append({"role": "assistant", "content": msg.content, "tool_calls": tool_calls_payload})
        updated_messages.append({"role": "assistant", "content": msg.content, "tool_calls": tool_calls_payload})

        for tc in msg.tool_calls:
            try:
                args = json.loads(tc.function.arguments)
            except json.JSONDecodeError:
                args = {}
            result = execute_tool(tc.function.name, args, db)
            result_str = json.dumps(result)

            tool_msg = {"role": "tool", "tool_call_id": tc.id, "content": result_str}
            conv.append(tool_msg)
            updated_messages.append(tool_msg)

            if log_callback:
                log_callback("tool_result", f"{tc.function.name}: {result_str[:200]}")

    # Max rounds exceeded — return empty
    return "", updated_messages
```
